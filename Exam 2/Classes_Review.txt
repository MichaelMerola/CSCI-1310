#include <iostream>

using namespace std;

class Fraction{
private:
    int numerator;
    int denomenator;
public:
    Fraction(int n, int d){
        numerator = n;
        denomenator = d;
    }
    void printValue(){
        float value = (float)numerator / (float)denomenator;
        cout<<value<<endl;
    }
    bool friend operator==(const Fraction &f1, const Fraction &f2){
        if(f1.numerator == f2.numerator && f1.numerator == f2.numerator){
            return true;
        }
        return false;
    }
    bool friend operator <(const Fraction &f1, const Fraction &f2){
        float val1 = f1.numerator / (float)f1.denomenator;
        float val2 = f2.numerator / (float)f2.denomenator;
        if(val1 < val2){
            return true;
        }
        return false;
    }
};
/*Design a class to represent a CU class. What are the variables for the class?
Ex: what would you need to represent this class.
Name, Enrollment max, currently enrolled, Description, Waitlist number,
Methods to change enrollment, such as drop, enroll.
To consider: restrictions on currently enrolled and waitlist.
*/
class CUClass{
private:
    string name;
    int maxEnrollment;
    int currentEnrollment;
    int waitlisted;
public:
    CUClass(string n, int maxEnroll){
        name = n;
        waitlisted = 0;
        currentEnrollment = 0;
        if(maxEnroll > 0){
            maxEnrollment = maxEnroll;
            cout<<"max"<<maxEnrollment<<endl;
        }else{
            cout<<"invalid max enrollment"<<endl;
            maxEnrollment = 0;
        }

    }
    void enrollStudent(){
        if(currentEnrollment < maxEnrollment){
            currentEnrollment++;
            cout<<"current: "<<currentEnrollment<<endl;
        }else{
            waitlisted++;
            cout<<"waitlist: "<<waitlisted<<endl;
        }
    }
    void dropStudent(){
        if(waitlisted > 0){
            waitlisted--;
        }else{
            currentEnrollment--;
            cout<<currentEnrollment<<endl;
        }
    }
    void printClassData(){
        cout<<"name: "<<name<<endl;
        cout<<"enrollment: "<<currentEnrollment<<endl;
        cout<<"waitlisted: "<<waitlisted<<endl;
    }

};
int main(){
    Fraction myFraction(5, 6);
    Fraction myFraction2(3, 6);
    myFraction.printValue();
    if(myFraction2 == myFraction){
        cout<<"equal"<<endl;
    }
    if(myFraction2 < myFraction){

    CUClass csci1310("csci1310", 150);

        csci1310.enrollStudent();

}
