int sumNonDuplicates(int arr[10]) {

    int sum = 0;
    bool isEqual;

    for (int i = 0; i < 10; i++) {

        isEqual = false; // reset for new iteration

        for(int j = 0; j < 10; j++){

            if (arr[i] == arr[j] && i!=j) {
                isEqual = true;
                break;
            }
        }

    if(!isEqual)
        sum += arr[i];

    }

    return sum;
}
