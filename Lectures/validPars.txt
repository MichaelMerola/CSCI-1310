int validParentheses (string str)
{
    int length = str.length();
    int rightPar = 0;
    int leftPar = 0;
    
    for (int i = 0; i < length; i++)
    {
        if (str[i] == '(' && i != length)
        {
            leftPar++;
        }
        else if (str[i] == ')' && i != 0)
        {
            rightPar++;
        }
    }
    
    if (rightPar == leftPar)
    {
        return 1;
    }
    else
    {
        return 0;
    }
    
}
